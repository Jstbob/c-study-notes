    16. 如何在 C 中将 2D 数组做为参数传递？

    在 C 中使用二维数组做参数传递：

    1. 两个维度是常量时

    #include <stdio.h>

const int ROWS = 2;
const int COLS = 3;

void printfArray(int array[ROWS][COLS]) {
  for (int i = 0; i < ROWS; ++i) {
    for (int j = 0; j < COLS; ++j) {
      printf("%d\t", array[i][j]);
    }
    printf("\n");
  }
}

int main() {
  int array[ROWS][COLS] = {{1, 2, 3}, {4, 5, 6}};
  printfArray(array);
}

    输出：

    1       2       3
4       5       6

    2. 第二个维度是常量时

    #include <stdio.h>



const int COLS = 3;



void printArray(int r,int array[][COLS]){

    for(int i=0;i<r;++i){

        for(int j=0;j<COLS;++j){

            printf("%d\t",array[i][j]);

        }

        printf("\n");

    }

}



int main(){

    int arr[][3]={{1,2,3},{4,5,6}};

    printArray(2, arr);

}

    输出：

    1       2       3
4       5       6

    3. 使用 VLA 可变数组（优先使用）

    如果编译器是和 C99 兼容，则可以通过可变数组传递参数。

    #include <stdio.h>



void printArray(int m, int n, int array[m][n]) {

  for (int i = 0; i < m; ++i) {

    for (int j = 0; j < n; ++j) {

      printf("%d\t", array[i][j]);

    }

    printf("\n");

  }

}



int main() {

  int array[2][3] = {{1, 2, 3}, {4, 5, 6}};

  printArray(2, 3, array);

}

    输出：

    1       2       3
4       5       6

    4. 使用单指针

    #include <stdio.h>



void printArray(int m, int n, int *array) {

  for (int i = 0; i < m; ++i) {

    for (int j = 0; j < n; ++j) {

      printf("%d\t", array[i * m + j]);

    }

    printf("\n");

  }

}



int main() {

  int array[2][3] = {{1, 2, 3}, {4, 5, 6}};

  printArray(2, 3, (int *)&array);

}

    输出：

    1       2       3
3       4       5
